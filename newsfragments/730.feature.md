Allow using complex type as array element type.

Example:
```
contract Foo {
    pub fn bar() -> i256 {
        let my_array: Array<Pair, 3> = [Pair::new(1, 0), Pair::new(2, 0), Pair::new(3, 0)]

        let sum: i256 = 0
        for pair in my_array {
            sum += pair.x
        }

        return sum
    }
}

struct Pair {
    pub x: i256
    pub y: i256
    
    pub fn new(_ x: i256, _ y: i256) -> Pair {
        return Pair(x, y)
    }
}
```
